```javascript
  // 初始化给所有输入框绑定回车聚焦事件
  initEnter = () => {
    // 需要在dom生成之后 才能获取到
    let items = document.querySelectorAll('#demoEnter')
    let activeList = []
    items.forEach(item => {
      if(!item.disabled){
        activeList.push(item)
      }
    })

    this.setState({
      activeList:activeList
    },() => {
      for (let i = 0; i < activeList.length; i++) {
        activeList[i].addEventListener('keydown', this.focus(i), false); // 每个input添加键盘监听
      }
    })
  }

  focus = (i) => {
    let { activeList } = this.state
    return (event) => {
      let next = i + 1 < activeList.length ? i + 1 : 0;
      const eve = event ? event : window.event;
        if (eve.keyCode === 13) {
          if (activeList[next].tagName === 'INPUT' || activeList[next].tagName === 'SELECT') {
            activeList[next] && activeList[next].focus();
          }
        }
    }
  }

  // 组件销毁 取消事件监听
  destory = () => {
    let { activeList } = this.state
    for (let i = 0; i < activeList.length; i++) {
      activeList[i].removeEventListener('keydown', this.focus(i), false); // 解除监听
    }
  }

  componentWillUnmount(){
    this.destory()
  }
```
